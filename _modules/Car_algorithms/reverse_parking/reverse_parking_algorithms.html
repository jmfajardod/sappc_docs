[docs]def rotationMatrixToEulerAngles(R):
    """
        Function to calculate the euler angles from a rotation matrix.

        :param R: Rotation matrix
        :type R: numpy.array

        :return: Euler angles in radians
        :rtype: numpy.array
    """

    sy = math.sqrt(R[0, 0] * R[0, 0] + R[1, 0] * R[1, 0])

    singular = sy < 1e-6

    if not singular:
        x = math.atan2(R[2, 1], R[2, 2])
        y = math.atan2(-R[2, 0], sy)
        z = math.atan2(R[1, 0], R[0, 0])
    else:
        x = math.atan2(-R[1, 2], R[1, 1])
        y = math.atan2(-R[2, 0], sy)
        z = 0

    return np.array([x, y, z])

[docs]def sg_filter(x, m, k=0):
    """
        Function to calculate the Savitzky-Golay filter coefficients.

        :param x: Vector of sample times
        :type x: numpy.array

        :param m: Order of the smoothing polynomial
        :type m: int

        :param k: Order of derivative
        :type k: int

        :return: Filter coefficients
        :rtype: numpy.array
    """

    mid = int(len(x) / 2)        
    a = x - x[mid]
    expa = lambda x: map(lambda i: i**x, a)
    exp_data = list(range(0,m+1))
    A = map(expa, exp_data)
    A = list(A)
    for i in range(len(A)):
        row_data = list(A[i])
        if i == 0:
            a_array = [np.array(row_data)]
        else:
            a_array.append(np.array(row_data))
    a_array = np.array(a_array)
    a_array = a_array.transpose()
    Ai = np.linalg.pinv(a_array)

    return Ai[k]



[docs]def smooth(x, y, size=5, order=2, deriv=0):
    """
        Function to calculate a derivative using the Savitzky-Golay filter.

        :param x: Vector of sample times
        :type x: numpy.array

        :param y: Vector of sample values
        :type y: numpy.array

        :param size: Number of samples to use for smoothing. Default is 5.
        :type size: int

        :param order: Order of the smoothing polynomial. Default is 2.
        :type order: int

        :param deriv: Order of the derivative. Default is 0.
        :type deriv: int

        :return: Filtered vector of sample values
        :rtype: numpy.array
    """

    if deriv > order:
        raise Exception("deriv must be <= order")

    n = len(x)
    m = int(size)

    result = np.zeros(n)

    for i in range(m, n-m):
        start, end = i - m, i + m + 1
        f = sg_filter(x[start:end], order, deriv)
        result[i] = np.dot(f, y[start:end])

    if deriv > 1:
        result *= math.factorial(deriv)

    return result




[docs]def get_Vel(time, pos, size, order):
    """
        Function to calculate the velocity of the vehicle

        :param time: Vector of sample times
        :type time: numpy.array

        :param pos: Vector of sample values
        :type pos: numpy.array

        :param size: Number of samples to use for smoothing.
        :type size: int

        :param order: Order of the smoothing polynomial.
        :type order: int

        :return: Estimated velocity from the position vector and the time vector
        :rtype: numpy.array
    """

    params = (time, pos, size, order)

    return smooth(*params, deriv=1)

        :param video_csv_path: Path to the video csv file
        :type video_csv_path: str

        :param video_path: Path to the video file
        :type video_path: str

        :param camera_matrix: Camera matrix
        :type camera_matrix: numpy.array

        :param dist_coeffs: Distortion coefficients
        :type dist_coeffs: numpy.array

        :param vel_stopped: Velocity threshold to consider the vehicle stopped

        :type view_image: bool
    """

    def __init__(self, video_csv_path, video_path, camera_matrix, dist_coefs, vel_stopped, view_image=False):

        self.video_csv = pd.read_csv(video_csv_path)
        self.video_path = video_path
        self.camera_matrix = camera_matrix
        self.dist_coefs = dist_coefs
        self.video_cap = None
        self.vel_stopped = vel_stopped
        self.view_image = view_image

        # arUco parameters
        self.arucoDict = cv2.aruco.Dictionary_get( cv2.aruco.DICT_4X4_50)  # Set aruco dictionary
        self.arucoParams = cv2.aruco.DetectorParameters_create()

        # Mask for allowed area
        self.mask_front = None
        self.mask_back = None

        # Dict for storing the results
        self.results = None
