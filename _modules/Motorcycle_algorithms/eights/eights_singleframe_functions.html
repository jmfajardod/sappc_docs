        Function to load image from file to GPU.

        :param imfile: path to the image file

        :return: image in GPU
        :rtype: torch.Tensor
    """

[docs]class EightsSingleFrameChecker:
    """
        Class to check if the vehicle gets out of the allowed area. It uses the optical flow to check if the vehicle is
        moving in the allowed area.

        :param video_csv_path: path to the csv file with the video information
        :type video_csv_path: str

        :param video_path: path to the video file
        :type video_path: str

        :param camera_matrix: camera matrix
        :type camera_matrix: np.array

        :param dist_coeffs: distortion coefficients
        :type dist_coeffs: np.array

        :param post: post number. From 17 to 22.
        :type post: int

        :param view_image: if True, it shows the image with the optical flow
        :type view_image: bool

        :return: None
    """

    def __init__(self, video_csv_path, video_path, camera_matrix, dist_coefs, post=17, view_image=False):

        # Define model parameters
        current_file_path = os.path.dirname(os.path.realpath(__file__))
        motorcycle_vision_algorithms_path = current_file_path.replace("eights", "")
        model_path = motorcycle_vision_algorithms_path + "RAFT/models/raft-small.pth"
        args = argparse.Namespace(mixed_precision=False, model=model_path, model_name='RAFT', path='images', small=True)

        print(args)

        # Load RAFT model
        self.model = torch.nn.DataParallel(RAFT(args))
        self.model.load_state_dict(torch.load(args.model))

        self.model = self.model.module
        self.model.to(DEVICE)
        self.model.eval()

        # Load camera parameters
        self.video_csv = pd.read_csv(video_csv_path)
        self.video_path = video_path
        self.camera_matrix = camera_matrix
        self.dist_coefs = dist_coefs
        self.video_cap = None
        self.post_num = post
        self.view_image = view_image

        # arUco parameters
        self.arucoDict = cv2.aruco.Dictionary_get( cv2.aruco.DICT_4X4_50)  # Set aruco dictionary
        self.arucoParams = cv2.aruco.DetectorParameters_create()

        # Mask for allowed area
        self.mask = None

        # Dict for storing the results
        self.results = None

        if post == 17:
            self.area_allowed = MASK_POST17
        elif post == 18:
            self.area_allowed = MASK_POST18
        elif post == 19:
            self.area_allowed = MASK_POST19
        elif post == 20:
            self.area_allowed = MASK_POST20
        elif post == 21:
            self.area_allowed = MASK_POST21
