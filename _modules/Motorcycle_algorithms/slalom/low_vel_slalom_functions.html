[docs]def load_image(imfile):
    """
        Function to load image from file to GPU.

        :param imfile: path to the image file

        :return: image in GPU
        :rtype: torch.Tensor
    """

    img = np.array(Image.open(imfile)).astype(np.uint8)
    img = torch.from_numpy(img).permute(2, 0, 1).float()
    return img[None].to(DEVICE)



[docs]def load_array_gpu(img):
    """
        Function to load array to GPU.

        :param img: array to load

        :return: array in GPU
        :rtype: torch.Tensor
    """

    img = np.array(img).astype(np.uint8)
    img = torch.from_numpy(img).permute(2, 0, 1).float()
    return img[None].to(DEVICE)



[docs]def viz(img, flo):
    """
        Function to visualize the optical flow.

        :param img: image
        :type img: torch.Tensor

        :param flo: optical flow
        :type flo: torch.Tensor

        :return: None
        :rtype: None
    """

    img = img[0].permute(1,2,0).cpu().numpy()
    flo = flo[0].permute(1,2,0).cpu().numpy()
    
    # map flow to rgb image
    flo = flow_viz.flow_to_image(flo)
    img_flo = np.concatenate([img, flo], axis=0)

    # import matplotlib.pyplot as plt
    # plt.imshow(img_flo / 255.0)
    # plt.show()

    cv2.imshow('image', img_flo[:, :, [2,1,0]]/255.0)
    cv2.waitKey()



[docs]class SlalomLowVelocityChecker:
    """
        Class to check if the vehicle has done the slalom at low velocity in the correct way. To estimate the trajectory of the vehicle, the optical flow is used.
        The position of the cones is manually set in the code.

        :param video_post_csv_path: path to the csv file with the video information
        :type video_post_csv_path: str

        :param video_post_path: path to the video file
        :type video_post_path: str

        :param camera_matrix: camera matrix
        :type camera_matrix: np.array

        :param dist_coefs: distortion coefficients
        :type dist_coefs: np.array

        :param post: post number. It can be 12, 13, 14 or 15
        :type post: int

        :param view_image: if True, it shows the image with the optical flow
        :type view_image: bool

        :return: None
    """

    def __init__(self, video_post_csv_path, video_post_path,
                camera_matrix, dist_coefs, post=1, view_image=False):

        # Define model parameters
        current_file_path = os.path.dirname(os.path.realpath(__file__))
        motorcycle_vision_algorithms_path = current_file_path.replace("slalom", "")
        model_path = motorcycle_vision_algorithms_path + "RAFT/models/raft-small.pth"
        args = argparse.Namespace(mixed_precision=False, model=model_path, model_name='RAFT', path='images', small=True)

        print(args)

        # Load RAFT model
        self.model = torch.nn.DataParallel(RAFT(args))
        self.model.load_state_dict(torch.load(args.model))

        self.model = self.model.module
        self.model.to(DEVICE)
        self.model.eval()

        # Open CSV
        self.video_post_csv = pd.read_csv(video_post_csv_path)
        self.video_post_csv_slalom = self.video_post_csv.copy()
        self.post = post
        print('Post: {}'.format(self.post))

        # Open video
        self.video_post_path = video_post_path
        self.video_post = cv2.VideoCapture(video_post_path + self.video_post_csv.iloc[0]['Video'])

        # Set camera properties
        frame_width = 1920
        frame_height = 1080
        video_fps = 15
        
        self.final_width = 852  # For low res RAFT
        self.final_height = 480 # For low res RAFT
        self.final_dim = (self.final_width, self.final_height)

        # Set cone position based on post
        self.post_num = post
        if post == 15:
            self.cone_pos_x = CONE_POS_X_15
            self.cone_pos_y = CONE_POS_Y_15
            self.cone_number = 1
        if post == 14:
            self.cone_pos_x = CONE_POS_X_14
            self.cone_pos_y = CONE_POS_Y_14
            self.cone_number = 2
        if post == 13:
            self.cone_pos_x = CONE_POS_X_13
            self.cone_pos_y = CONE_POS_Y_13
            self.cone_number = 3
        if post == 12:
            self.cone_pos_x = CONE_POS_X_12
            self.cone_pos_y = CONE_POS_Y_12
            self.cone_number = [4, 5]

        # Save camera matrix and distortion coefficients
        self.camera_matrix = camera_matrix
        self.dist_coefs = dist_coefs
        self.newcameramtx = cv2.getOptimalNewCameraMatrix(camera_matrix, dist_coefs, (1920, 1080), 1, (1920, 1080))[0]

        # Save view image
        self.view_image = view_image

        # Set empty dictionary for results
        self.results = None

        # Open VideoWriter
        video_fourcc = cv2.VideoWriter_fourcc('m', 'p', '4', 'v')  # Best
        self.out_slalom = cv2.VideoWriter(video_post_path + 'proccessed_slalom.avi',video_fourcc, video_fps, (self.final_width, self.final_height))


[docs]    def process_video(self):
        """
            Function to check if the vehicle has done the slalom at low velocity in the correct way.

            :return: None. The results are saved in a video and a csv file.
            :rtype: None
        """

        print("Processing video...")

        # init_frame = 2300
        # end_frame = 3850
        # for ii in range(init_frame):
        #     ret_vid, frame_1 = self.video_post.read()

        frame_1 = None
        frame_2 = None

        current_pos_x = None
        current_pos_y = None
        prev_pos_x = None
        prev_pos_y = None

        crossed_cone = False
        crossed_cone_dir = None
        maintain_text_cone = 30
        counter_maintain_text = 0

        pbar = tqdm(range(len(self.video_post_csv)))
        for ii in pbar:

            pbar.set_description("Processing high velocity slalom post %i:  %i" % (self.post_num, ii))

            if frame_1 is None:
                ret_vid, frame_1 = self.video_post.read()
                frame_1 = cv2.resize(frame_1, self.final_dim, interpolation = cv2.INTER_AREA)
                frame_1_rgb = cv2.cvtColor(frame_1, cv2.COLOR_BGR2RGB)
                continue
            
            # Read frame
            current_time = self.video_post_csv_slalom.iloc[ii]['Time']
            ret_vid, frame_2 = self.video_post.read()

            if ret_vid:

                # Convert to RGB
                frame_2 = cv2.resize(frame_2, self.final_dim, interpolation = cv2.INTER_AREA)
                frame_2_rgb = cv2.cvtColor(frame_2, cv2.COLOR_BGR2RGB)

                # Compute optical flow
                with torch.no_grad():
                    # Pass images to GPU
                    image1 = load_array_gpu(frame_1_rgb.copy())
                    image2 = load_array_gpu(frame_2_rgb.copy())

                    # Pad images according to model input size
                    padder = InputPadder(image1.shape)
                    image1, image2 = padder.pad(image1, image2)

                    # Compute optical flow
                    flow_low, flow_up = self.model(image1, image2, iters=20, test_mode=True)
                    del image1, image2

                    # Obtain flow images
                    flow_up_cpu  = flow_up[0].permute(1, 2, 0).cpu().numpy()    # High resolution flow

                    # Compute abs optical flow and normalize
                    flow_mag = np.sqrt(np.square(flow_up_cpu[:,:,0]) + np.square(flow_up_cpu[:,:,1]))
                    flow_mag = np.uint8( (flow_mag / flow_mag.max()) * 255 )

                    # Threshold optical flow
                    flow_mag_thresh = cv2.threshold(flow_mag, 130, 255, cv2.THRESH_BINARY)[1]

                    # Show image
                    if self.view_image:
                        im_thresholded = cv2.resize(flow_mag_thresh, (0, 0), fx=1.0, fy=1.0)
                        cv2.imshow('im_thresholded', im_thresholded)
                        keypressed = cv2.waitKey(1)
                        if keypressed == ord('q'):
                            break

                    #- Get center of mass of optical flow
                    cnts = cv2.findContours(flow_mag_thresh.copy(), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
                    cnts = imutils.grab_contours(cnts)
                    # print("Length contours init: ", len(cnts))

                    #- Filter contours by area
                    new_cnts = []
                    for contour in cnts:
                        area = cv2.contourArea(contour)
                        if area > 3000 and area < 50000:
                            new_cnts.append(contour)
                    cnts = new_cnts
                    # print("Length contours after: ", len(cnts))

                    #- Update dataframe
                    if self.results is None:
                        if self.post == 12:
                            self.results = {"Time": [], "Pos_X": [], "Pos_Y": [], "Crossed_cone_"+str(self.cone_number[0]): [], "Crossed_cone_"+str(self.cone_number[1]): []}
                        else: 
                            self.results = {"Time": [], "Pos_X": [], "Pos_Y": [], "Crossed_cone_"+str(self.cone_number): []}

                    #- Set image to display
                    img_display = frame_1.copy()
                    
                    if self.post == 12:
                        for ii in range(len(self.cone_number)):
                            img_display = cv2.circle(img_display, (self.cone_pos_x[ii], self.cone_pos_y[ii]), 15, (63, 67, 255), -1) # Cone
                    else:
                        img_display = cv2.circle(img_display, (self.cone_pos_x, self.cone_pos_y), 15, (63, 67, 255), -1) # Cone
                        #img_display = cv2.line(img_display, (self.cone_pos_x,0), (self.cone_pos_x,self.final_height), (0,0,255), thickness=2)

                    if len(cnts) == 0:
                        self.results["Time"].append(current_time)
                        self.results["Pos_X"].append(None)
                        self.results["Pos_Y"].append(None)
                        if self.post == 12:
                            self.results["Crossed_cone_"+str(self.cone_number[0])].append(None)
                            self.results["Crossed_cone_"+str(self.cone_number[1])].append(None)
                        else:
                            self.results["Crossed_cone_"+str(self.cone_number)].append(None)
                    else:
                        #- Get contour with largest area
                        contour = max(cnts, key=cv2.contourArea)
                        #print("Area of contour:  ", cv2.contourArea(contour))

                        #- Compute the center of the contour
                        Moments = cv2.moments(contour)
                        cX = int(Moments["m10"] / Moments["m00"])
                        cY = int(Moments["m01"] / Moments["m00"])

                        self.results["Time"].append(current_time)
                        self.results["Pos_X"].append(cX)
                        self.results["Pos_Y"].append(cY)

                        if prev_pos_x is None:
                            prev_pos_x = cX
                            prev_pos_y = cY

                        current_pos_x = cX
                        current_pos_y = cY

                        if self.post == 12:
                            #- First cone
                            if current_pos_x >= self.cone_pos_x[0] and prev_pos_x <= self.cone_pos_x[0]:
                                crossed_cone = True
                                crossed_cone_number = 0
                                counter_maintain_text = 0
                                if current_pos_y > self.cone_pos_y[0]:
                                    crossed_cone_dir = "Right"
                                else:
                                    crossed_cone_dir = "Left"
                                self.results["Crossed_cone_"+str(self.cone_number[0])].append(crossed_cone_dir)
                            else:
                                self.results["Crossed_cone_"+str(self.cone_number[0])].append(None)
                            # Second cone
                            if current_pos_x >= self.cone_pos_x[1] and prev_pos_x <= self.cone_pos_x[1]:
                                crossed_cone = True
                                crossed_cone_number = 1
                                counter_maintain_text = 0
                                if current_pos_y > self.cone_pos_y[1]:
                                    crossed_cone_dir = "Right"
                                else:
                                    crossed_cone_dir = "Left"
                                self.results["Crossed_cone_"+str(self.cone_number[1])].append(crossed_cone_dir)
                            else:
                                self.results["Crossed_cone_"+str(self.cone_number[1])].append(None)

                        else:
                            if current_pos_x >= self.cone_pos_x and prev_pos_x <= self.cone_pos_x:
                                crossed_cone = True
                                counter_maintain_text = 0
                                if current_pos_y > self.cone_pos_y:
                                    crossed_cone_dir = "Right"
                                else:
                                    crossed_cone_dir = "Left"
                                self.results["Crossed_cone_"+str(self.cone_number)].append(crossed_cone_dir)
                            else:
                                self.results["Crossed_cone_"+str(self.cone_number)].append(None)

                        prev_pos_x = current_pos_x
                        prev_pos_y = current_pos_y
                    
                        # Draw the contour and center of the shape on the image
                        img_display = cv2.drawContours(img_display, [contour], -1, (0, 255, 0), 2)
                        img_display =cv2.circle(img_display, (cX, cY), 7, (255, 0, 0), -1)
                        img_display =cv2.putText(img_display, "center", (cX - 20, cY - 20), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)
                        str_pos = 'X: {:2.0f} Y: {:2.0f} '.format(cX, cY)
                        img_display = cv2.putText(img_display, str_pos, (600,450) , cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 0, 0), 2, cv2.LINE_AA)

                    if crossed_cone:
                        if self.post == 12:
                            #- First cone
                            if crossed_cone_number == 0:
                                if crossed_cone_dir == "Right":
                                    img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x[0]-100,self.cone_pos_y[0]+100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                                else:
                                    img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x[0]-100,self.cone_pos_y[0]-100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                                counter_maintain_text += 1
                                if counter_maintain_text > maintain_text_cone:
                                    crossed_cone = False
                            #- Second cone
                            elif crossed_cone_number == 1:
                                if crossed_cone_dir == "Right":
                                    img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x[1]-300,self.cone_pos_y[1]+100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                                else:
                                    img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x[1]-300,self.cone_pos_y[1]-100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                                counter_maintain_text += 1
                                if counter_maintain_text > maintain_text_cone:
                                    crossed_cone = False


                        else:
                            if crossed_cone_dir == "Right":
                                img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x-300,self.cone_pos_y+100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                            else:
                                img_display =cv2.putText(img_display, "Crossed cone " + crossed_cone_dir, (self.cone_pos_x-300,self.cone_pos_y-100-20), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (0, 255, 255), 2)
                            
                            counter_maintain_text += 1
                            if counter_maintain_text > maintain_text_cone:
                                crossed_cone = False

                    # Save frame in video
                    self.out_slalom.write(img_display)

                    # Show image
                    if self.view_image:
                        im_out_area = cv2.resize(img_display, (0, 0), fx=1.0, fy=1.0)
                        cv2.imshow('image', im_out_area)
                        keypressed = cv2.waitKey(1)
                        if keypressed == ord('q'):
                            break

                    # Update frame
                    frame_1 = frame_2
                    frame_1_rgb = frame_2_rgb

        # Close video
        self.video_post.release()
        self.out_slalom.release()
        print("Close video")

        # Convert results to dataframe
        self.results = pd.DataFrame(self.results)
        self.results.to_csv(self.video_post_path + "slalom_results.csv", index=False)
